  3 var SocketIoClientTransport = function (host, options) {
  4   this._socket = new io.Socket(host, options);
  5   this._callbacks = null;
  6 };
  7 SocketIoClientTransport.prototype = {
  8   connect: function (callbacks) {
  9     this._socket.connect();
 10     this._callbacks = callbacks;
 11     this._socket.addEvent('connect', curry(this.onConnect, this));
 12     this._socket.addEvent('disconnect', curry(this.onDisconnect, this));
 13     //this._socket.addEvent('close', curry(this.onClose, this));
 14     this._socket.addEvent('message', curry(this.onMessage, this));
 15   },
 16   onConnect: function () {
 17     this._callbacks.onConnect.call(this._callbacks, this);
 18   },
 19   onDisconnect: function () {
 20     this._callbacks.onDisconnect.call(this._callbacks, this);
 21   },
 22   onMessage: function (message) {
 23     this._callbacks.onMessage.call(this._callbacks, JSON.parse(message), this);
 24   },
 25   send: function (message) {
 26     this._socket.send(JSON.stringify(message));
 27   }
 28 }
