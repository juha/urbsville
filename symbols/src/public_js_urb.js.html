  2 function curry(fn, scope) {
  3   var __method = fn;
  4   var self = scope || window;
  5   var args = [];
  6   for (var i=2, len = arguments.length; i < len; ++i) {
  7     args.push(arguments[i]);
  8   }
  9   return function() {
 10     var a = [];
 11     for (var i in arguments) {
 12       a.push(arguments[i]);
 13     }
 14     for (var j in args) {
 15       a.push(args[j]);
 16     }
 17     return __method.apply(self, a);
 18   };
 19 }

 37 function clone(obj) {
 38   var newObj = (obj instanceof Array) ? [] : {};
 39   for (i in obj) {
 40     if (i == 'clone') continue;
 41     if (obj[i] && typeof obj[i] == "object") {
 42       newObj[i] = clone(obj[i]);
 43     } else {
 44       newObj[i] = obj[i];
 45     }
 46   }
 47   return newObj;
 48 }

 53 /**
 54  * @class Base class for objects that send or receive events.
 55  * @param {String} kind Like a class identifier
 56  * @param {String} name A unique identifier for this object within its class
 57  */
 58 var Evented = function (kind, name) {
 59   this._listeners = [];
 60   this._name = name;
 61   this._kind = kind;
 62 };
 63 Evented.prototype = {
 64   kind: function () { return this._kind; },
 65   name: function () { return this._name; },
 66   /**
 67    * Return what we hope is a unique identifier for this instance.
 68    *
 69    * Defaults to be kind + / + name.
 70    * @returns {String} Unique identifier for this instance
 71    */
 72   id: function () { return this.kind() + '/' + this.name(); },
 73   /**
 74    * Serialize this instance, usually for sending over the wire.
 75    * @returns {Object} A simple object of strings
 76    */
 77   toDict: function () {
 78     return {kind: this.kind(), name: this.name(), id: this.id()};
 79   },                        
 80   listeners: function () { return this._listeners; },
 81   /**
 82    * Add to this instance's list of listeners.
 83    * @param {implements Listener} listener
 84    */
 85   addListener: function (listener) {
 86     this._listeners.push(listener);
 87   },
 88   /**
 89    * Remove a listener from this instance's list of listeners.
 90    * @param {implements Listener} listener
 91    */
 92   removeListener: function (listener) {
 93     for (var i in this._listeners) {
 94       if (this._listeners[i] == listener) {
 95         this._listeners.splice(i, 1);
 96         break;
 97       }
 98     }
 99   },
100   /**
101    * Notify interested listeners about an event
102    * @param {Object} An object with a list of topics and some data
103    */
104   notifyListeners: function (event) {
105     if (event.topic.indexOf(this.id()) === -1) {
106       event.topic.unshift(this.id());
107     }
108     //sys.puts(this.id() + ' 1event.data: ' + event.data);
109     for (var i in this._listeners) {
110       var newEvent = clone(event);
111       //sys.puts(this.id() + ' 2event.data: ' + newEvent.data);
112       if (this._listeners[i].match(newEvent.topic)) {
113         //sys.puts(sys.inspect(newEvent));
114         this._listeners[i].send(newEvent);
115       }
116     }
117   }
118 };

121 var Listener = function (matcher, callback) {
122   this.matcher = matcher;
123   this.callback = callback;
124 };
125 Listener.prototype = {
126   match: function (topic) {
127     for (var i in topic) {
128       if (topic[i].match(this.matcher)) {
129         return true;
130       }
131     }
132     return false;
133   },
134   send: function (event) {
135     this.callback(event);
136   }

253 /**
254  * @constructor
255  */
256 var Device = function(kind, name, properties) {
257   Evented.call(this, kind, name);
258   if (!this._properties) {
259     this._properties = {};
260   }
261   if (properties) {
262     extend(this._properties, properties);
263   }
264 };
265 inherit(Device, Evented);
266 /** @lends Device.prototype */
267 extend(Device.prototype, {
268   properties: function () { return this._properties; },
269   toDict: function () {
270     var dict = Evented.prototype.toDict.call(this);
271     dict.properties = this.properties();
272     return dict;
273   },
274   getProperty: function (property) {
275     if (this._properties[property] === undefined) {
276       throw 'not a valid property: ' + property;
277     }
278     return this._properties[property];
279   },
280   setProperty: function (property, value) {
281     if (this._properties[property] === undefined) {
282       throw 'not a valid property: ' + property;
283     }
284     this._properties[property] = value;
285     var event = {topic: ['property/' + property],
286                  data: value};
287     this.notifyListeners(event);
288   }
289 });

292 /**
293  * @constructor
294  */
295 var Urb = function (kind, name) {
296   Evented.apply(this, arguments);
297   this._devices = [];
298 };
299 inherit(Urb, Evented);
300 /** @lends Urb.prototype */
301 extend(Urb.prototype, {
302   devices: function () { return this._devices; },
303   addDevice: function (device) {
304     this._devices.push(device);
305     
306     // by default we're going to listen to all updates from the device
307     device.addListener(this.deviceListener());
308     this.notifyListeners({topic: ['urb/deviceAdded'],
309                           data: device.toDict()});
310   },
311   removeDevice: function (device) {
312     for (var i in this._devices) {
313       if (this._devices[i] == device) {
314         device.removeListener(this.deviceListener());
315         this.notifyListeners({topic: ['urb/deviceRemoved'],
316                               data: device.toDict()});
317         this._devices.splice(i, 1);
318         break;
319       }
320     }
321   },
322   /* returns the device listener singleton */
323   deviceListener: function () {
324     if (this._deviceListener === undefined) {
325       var self = this;
326       this._deviceListener = new Listener(/.*/,
327                                           curry(this.notifyListeners, this));
328     }
329     return this._deviceListener; 
330   }
331 });

336 /**
337  * @class DeviceProxy for a Device being accessed remotely.
338  * Uses something of a hacked up mixin pattern to add Proxy methods.
339  *
340  * Instances of this class are used Client-side to represent Devices on a
341  * remote ApiServer.
342  * 
343  * Instances of this class are also used Server-side to represent Devices
344  * being provided to a DeviceServer by a remote DeviceClient.
345  *
346  * @param {String} kind {@link Evented}
347  * @param {String} name {@link Evented}
348  * @param {String} properties {@link Device}
349  * @param {implements Connection} connection {@link Proxy}
350  *
351  * @extends Device
352  */
353 var DeviceProxy = function (kind, name, properties, connection) {
354   Device.call(this, kind, name, properties);
355   Proxy.call(this, connection);
356 };
357 inherit(DeviceProxy, Device);
358 extend(DeviceProxy.prototype, Proxy.prototype);
359 /** @lends DeviceProxy.prototype */
360 extend(DeviceProxy.prototype, {
361   /**
362    * Update internal state and notify listeners when receiving remote state.
363    * @param {Object} event A simple event object.
364    */
365   onEvent: function (event) {
366     var matcher = /property\/(.*)/;
367     for (var i in event.topic) {
368       var match = matcher.exec(event.topic[i]);
369       if (match) {
370         this._setProperty(match[1], event.data[match[1]]);
371       }
372     }
373   },
374   /**
375    * Actually change the internal represenation and notify listeners
376    * @see Device#setProperty
377    * @private
378    */
379   _setProperty: function () {
380     Device.prototype.setProperty.apply(this, arguments);
381   },
382   /**
383    * Sends a setProperty RPC to the remote device.
384    *
385    * Note that the property will not actually be set until the remote
386    * device has acknowledged the RPC and set its own property.
387    *
388    * @see Device#setProperty
389    */
390   setProperty: function () {
391     this.rpc(this.id(), 'setProperty', arguments);
392   }
393 });

428 /**
429  * @class ApiClient Client-side API client, connects to an ApiServer.
430  *
431  * Builds a local representation with UrbProxy and DeviceProxy objects.
432  * 
433  * Provides a Connection interface for Proxy objects.
434  *
435  * @extends Evented
436  * @borrows Connection#send as this.send
437  */
438 var ApiClient = function () {
439   Evented.apply(this, arguments);
440   this._urb = null;
441   this.transport = null;
442 };
443 inherit(ApiClient, Evented);
444 /** @lends ApiClient.prototype */
445 extend(ApiClient.prototype, {
446   /**
447    * Initiate a connection using the given transport.
448    *
449    * TODO(termie): Should probably clear current state if any exists.
450    *
451    * @param {implements Transport} transport {@link Transport} 
452    */
453   connect: function (transport) {
454     this.transport = transport;
455     this.transport.connect(this)
456   },
457   /**
458    * Event handler for connect events. Notifies listeners.
459    */
460   onConnect: function () {
461     this.notifyListeners({topic: ['client/connect'],
462                           data: this.id()});
463   },
464   /**
465    * Event handler for disconnect events. Notifies listeners.
466    *
467    * TODO(termie): clean up urbs and devices
468    */
469   onDisconnect: function () {
470     this.notifyListeners({topic: ['client/disconnect'],
471                           data: this.id()});
472   },
473   /**
474    * Event handler for Urb events. Builds an UrbProxy, notifies listeners.
475    * 
476    * Also makes sure that further events from the remote ApiServer are
477    * forwarded to the UrbProxy.
478    *
479    * @param {Object} urb A serialized Urb. {@link Urb#toDict}
480    */
481   onUrb: function (urb) {
482     var proxy = new UrbProxy(urb.kind, urb.name, this);
483     this._urbs[proxy.id()] = proxy;
484     this.addListener(new Listener(new RegExp(proxy.id()),
485                                   curry(proxy.onEvent, proxy)));
486     this.notifyListeners({topic: ['client/urb'],
487                           data: proxy.id()});
488   },
489   /**
490    * Event handler for Device events. Builds a DeviceProxy, notifies listeners.
491    * 
492    * Also makes sure that further events from the remote ApiServer are
493    * forwarded to the DeviceProxy.
494    *  
495    * TODO(termie): I'd prefer for us to figure out the urb from somewhere
496    *               that isn't in the device because in other circumstances
497    *               there is no reason why the device would know which urb is
498    *               holding on to it. This is easily remedied if we make things
499    *               one urb per client but I am expecting to do access control
500    *               at the urb level rather than the client level.
501    *
502    * @param {Object} device A serialized Device. {@link Device#toDict}
503    */
504   onDevice: function (device) {
505     var proxy = new DeviceProxy(device.kind, 
506                                 device.name,
507                                 device.properties,
508                                 this);
509     this._urbs[device.urb].addDevice(DeviceProxy);
510     this.addListener(new Listener(new RegExp(proxy.id()),
511                                   curry(proxy.onEvent, proxy)));
512     this.notifyListeners({topic: ['client/device'],
513                           data: proxy.id()});
514   },
515   /**
516    * Generic event handler. Notify listeners.
517    *
518    * In many cases the event will be a state change event for a remote Urb
519    * or Device.
520    *
521    * @param {Object} event {@link Event}
522    */
523   onEvent: function (event) {
524     this.notifyListeners(event);
525   },
526   /**
527    * Handles receiving a new message from the remote ApiServer and routing it.
528    *
529    * @param {object} message {@link Message}
530    */
531   onMessage: function (message) {
532     if (message.kind == 'urb') {
533       this.onUrb(message.data);
534     } else if (message.kind == 'device') {
535       this.onDevice(message.data);
536     } else if (message.kind == 'event') {
537       this.onEvent(message.data);
538     }
539   }
540 });

545 /**
546  * @class ApiServer provides access to Urbs and their Devices
547  * @extends Evented
548  */
549 var ApiServer = function () {
550   Evented.apply(this, arguments);
551   this._clients = [];
552   this._urbs = [];
553   this.transport = null;
554   this.addListener(this.serverListener());
555 };
556 inherit(ApiServer, Evented);
557 /** @lends ApiServer.prototype */
558 extend(ApiServer.prototype, {
559   /**
560    * Initiate listening by the ApiServer using given Transport.
561    *
562    * @param {implements Transport} transport {@link Transport}
563    */
564   listen: function(transport) {
565     this.transport = transport;
566     this.transport.listen(this);
567   },
568   /**
569    * Add an Urb to the list of Urbs brokered by the ApiServer.
570    *
571    * Notify existing clients of the new Urb and all of its Devices.
572    *
573    * @param {Urb} urb An Urb.
574    */
575   addUrb: function (urb) {
576     urb.addListener(this.urbListener());
577     this._urbs.push(urb);
578     this.notifyClients({kind: 'urb',
579                         data: urb.toDict()});
580     
581     var devices = urb.devices();
582     for (var i in devices) {
583       this.notifyClients({kind: 'device',
584                           data: devices[i].toDict()});
585     }
586   },
587   /**
588    * Notifies listeners of the ApiServer on events from Urbs.
589    * @return {Listener} A listener singleton for events from Urbs 
590    */
591   urbListener: function () {
592     if (this._urbListener === undefined) {
593       var self = this;
594       this._urbListener = new Listener(/.*/, curry(this.notifyListeners, this));
595     }
596     return this._urbListener; 
597   },
598   /**
599    * Notifies clients on events from the ApiServer.
600    * @return {Listener} A listener singleton for events from the ApiServer 
601    */
602   serverListener: function () {
603     if (this._serverListener === undefined) {
604       var self = this;
605       this._serverListener = new Listener(/.*/, curry(this.onEvent, this));
606     }
607     return this._serverListener; 
608   },
609   /**
610    * Notify connected clients
611    *
612    * @param {Object} message {@link Message}
613    */
614   notifyClients: function (message) {
615     for (var i in this._clients) {
616       this._clients[i].send(message);
617     }
618   },
619   /**
620    * When a remote client connects keep track of it and notify listeners.
621    *
622    * @param {ApiClientProxy} client A Proxy for the remote ApiClient
623    */
624   onClientConnect: function (client) {
625     this._clients.push(client);
626     this.notifyListeners({topic: ['server/clientConnect'],
627                           data: client.id()});
628   },
629   /**
630    * Handle messages from the remote client.
631    * 
632    * At the moment the only handled Message kind is rpc.
633    *
634    * @param {Object} message {@link Message}
635    * @param {ApiClientProxy} client A Proxy for the remote ApiClient
636    */
637   onClientMessage: function (message, client) {
638     if (message.kind == 'rpc') {
639       this.onRpc(message.data, client);
640       this.notifyListeners({topic: ['server/rpc'], data: message.data});
641     }
642   },
643   /**
644    * When a remote client disconnects remove them and notify listeners.
645    *
646    * @param {ApiClientProxy} client A Proxy for a remote ApiClient.
647    */
648   onClientDisconnect: function (client) {
649     for (var i in this._clients) {
650       if (this._clients[i] == client) {
651         this._clients.splice(i, 1);
652         this.notifyListeners({topic: ['server/clientDisconnect'],
653                               data: client.id()});
654         break;
655       }
656     }
657   },
658   /**
659    * Handle an RPC call from an ApiClient.
660    *
661    * TODO(termie): this can probably be done a little more elegantly.
662    * TODO(termie): this can result in multiple calls if multiple devices
663    *               have the same id, not sure if that is desirable yet.
664    *
665    * @param {Object} rpc {@link Rpc}
666    * @param {ApiClientProxy} client A Proxy for a remote ApiClient.
667    */
668   onRpc: function (rpc, client) {
669     for (var u in this._urbs) {
670       var devices = this._urbs[u].devices();
671       for (var d in devices) {
672         if (rpc.id == devices[d].id()) {
673           devices[d][rpc.method].apply(devices[d], rpc.arguments);
674         }
675       }
676     }
677   },
678   /**
679    * Forwards Events from the Urbs and Devices to remote ApiClients
680    *
681    * @param {Object} event {@link Event}
682    */ 
683   onEvent: function (event) {
684     this.notifyClients({kind: 'event', data: event});
685   }
686 });

690  * @class Accepts connections from remote devices and publishes them locally
691  * @param {Urb} An Urb instance through which to publish the devices
692  */
693 var DeviceServer = function (kind, name, urb) {
694   Evented.call(this, kind, name);
695   this.urb = urb;
696   this._clients = {};
697 };
698 inherit(DeviceServer, Evented);
699 /** @lends DeviceServer */
700 extend(DeviceServer.prototype, {
701   /**
702    * Initiate listening by the DeviceServer. To be implemented by subclasses.
703    */

705     // Not Implemented
706   },
707   onClientConnect: function (client) {
708     this._clients[client.id()] = [];
709     this.notifyListeners({topic: ['deviceserver/clientConnect'],
710                           data: client.id()});
711   },
712   onClientDisconnect: function (client) {
713     for (var i in this._clients[client.id()]) {
714       this.urb.removeDevice(this._clients[client.id()][i]);
715     }
716     delete this._clients[client.id()];
717     this.notifyListeners({topic: ['deviceserver/clientDisconnect'],
718                           data: client.id()});
719   },
720   onClientMessage: function (message, client) {
721     if (message.kind == 'device') {
722       this.onDevice(message.data, client);
723     } else if (message.kind == 'event') {
724       this.onEvent(message.data);
725     }
726   },

728     //sys.puts(sys.inspect(device));
729     var proxy = new DeviceProxy(device.kind, 
730                                 device.name,
731                                 device.properties,
732                                 this);
733     this.addListener(new Listener(new RegExp(proxy.id()),
734                                   curry(proxy.onEvent, proxy)));
735     this._clients[client.id()].push(proxy);
736     this.urb.addDevice(proxy);
737     this.notifyListeners({topic: ['deviceserver/device'],
738                           data: proxy.id()});
739   },
740   onEvent: function (event) {
741     this.notifyListeners(event);
742   },
