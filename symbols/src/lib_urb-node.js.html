1 var http = require('http');
  2 var urb = require('urb');
  3 var io = require('socket.io');
  4 
  5 var curry = urb.curry;
  6 var extend = urb.extend;
  7 var inherit = urb.inherit;
  8 
  9 
 10 var SocketIoServerTransport = function (port, options) {
 11   this._port = port;
 12   this._options = options;
 13   this._server = null;
 14   this._clients = {};
 15   this._callbacks = null;
 16 };
 17 SocketIoServerTransport.prototype = {
 18   listen: function (callbacks) {
 19     this._callbacks = callbacks;
 20     this._server = http.createServer(function () { });
 21     this._server.listen(this._port);
 22     this._socket = io.listen(this._server, this._options);
 23 
 24 
 25     // NOTE(termie): SocketIO's addListener has a different signature from ours
 26     this._socket.addListener('clientConnect',
 27                              curry(this.onClientConnect, this));
 28     this._socket.addListener('clientMessage',
 29                              curry(this.onClientMessage, this));
 30     this._socket.addListener('clientDisconnect',
 31                              curry(this.onClientDisconnect, this));
 32   },
 33   _getClientIndex: function (client) {
 34     return client.sessionId;
 35   },
 36   onClientConnect: function (client) {
 37     var idx = this._getClientIndex(client);
 38     var myClient = new SocketIoServerTransportClient(idx, client);
 39     this._clients[idx] = myClient;
 40     this._callbacks.onClientConnect.call(this._callbacks, myClient);
 41   },
 42   onClientMessage: function (message, client) {
 43     var idx = this._getClientIndex(client);
 44     var myClient = this._clients[idx];
 45     this._callbacks.onClientMessage.call(this._callbacks, message, myClient);
 46   },
 47   onClientDisconnect: function (client) {
 48     var idx = this._getClientIndex(client);
 49     var myClient = this._clients[idx];
 50     this._callbacks.onClientDisconnect.call(this._callbacks, myClient);
 51     delete this._clients[idx];
 52   }
 53 };
 54 
 55 SocketIoServerTransportClient = function (name, client) {
 56   urb.Evented.call(this, 'SocketIoServerTransportClient', name);
 57   this._client = client;
 58 }
 59 inherit(SocketIoServerTransportClient, urb.Evented);
 60 extend(SocketIoServerTransportClient.prototype, {
 61   send: function (message) { this._client.send(JSON.stringify(message)); }
 62 });
 63 
 64 exports.SocketIoServerTransport = SocketIoServerTransport;
